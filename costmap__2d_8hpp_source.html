504 template<class ActionType>
505 inline void raytraceLine(
506 ActionType at, unsigned int x0, unsigned int y0, unsigned int x1,
507 unsigned int y1,
508 unsigned int max_length = UINT_MAX, unsigned int min_length = 0)
509 {
510 int dx_full = x1 - x0;
511 int dy_full = y1 - y0;
512
513 // we need to chose how much to scale our dominant dimension,
514 // based on the maximum length of the line
515 double dist = std::hypot(dx_full, dy_full);
516 if (dist < min_length) {
517 return;
518 }
519
520 unsigned int min_x0, min_y0;
521 if (dist > 0.0) {
522 // Adjust starting point and offset to start from min_length distance
523 min_x0 = (unsigned int)(x0 + dx_full / dist * min_length);
524 min_y0 = (unsigned int)(y0 + dy_full / dist * min_length);
525 } else {
526 // dist can be 0 if [x0, y0]==[x1, y1].
527 // In this case only this cell should be processed.
528 min_x0 = x0;
529 min_y0 = y0;
530 }
531 unsigned int offset = min_y0 * size_x_ + min_x0;
532
533 int dx = x1 - min_x0;
534 int dy = y1 - min_y0;
535
536 unsigned int abs_dx = abs(dx);
537 unsigned int abs_dy = abs(dy);
538
539 int offset_dx = sign(dx);
540 int offset_dy = sign(dy) * size_x_;
541
542 double scale = (dist == 0.0) ? 1.0 : std::min(1.0, max_length / dist);
543 // if x is dominant
544 if (abs_dx >= abs_dy) {
545 int error_y = abs_dx / 2;
546
547 bresenham2D(
548 at, abs_dx, abs_dy, error_y, offset_dx, offset_dy, offset, (unsigned int)(scale * abs_dx));
549 return;
550 }
551
552 // otherwise y is dominant
553 int error_x = abs_dy / 2;
554
555 bresenham2D(
556 at, abs_dy, abs_dx, error_x, offset_dy, offset_dx, offset, (unsigned int)(scale * abs_dy));
557 }
558
559private:
564 template<class ActionType>
565 inline void bresenham2D(
566 ActionType at, unsigned int abs_da, unsigned int abs_db, int error_b,
567 int offset_a,
568 int offset_b, unsigned int offset,
569 unsigned int max_length)
570 {
571 unsigned int end = std::min(max_length, abs_da);
572 for (unsigned int i = 0; i < end; ++i) {
573 at(offset);
574 offset += offset_a;
575 error_b += abs_db;
576 if ((unsigned int)error_b >= abs_da) {
577 offset += offset_b;
578 error_b -= abs_da;
579 }
580 }
581 at(offset);
582 }
