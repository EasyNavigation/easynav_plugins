120 inline unsigned char computeCost(double distance) const
121 {
122 unsigned char cost = 0;
123 if (distance == 0) {
124 cost = LETHAL_OBSTACLE;
125 } else if (distance * resolution_ <= inscribed_radius_) {
126 cost = INSCRIBED_INFLATED_OBSTACLE;
127 } else {
128 // make sure cost falls off by Euclidean distance
129 double factor =
130 exp(-1.0 * cost_scaling_factor_ * (distance * resolution_ - inscribed_radius_));
131 cost = static_cast<unsigned char>((INSCRIBED_INFLATED_OBSTACLE - 1) * factor);
132 }
133 return cost;
134 }
